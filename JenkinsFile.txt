node("docker") {
    docker.withRegistry('dockerReg', '') {
    
        git url: "https://github.com/enavaku/REPOTraining.git", credentialsId: 'enavaku'
    
        sh "git rev-parse HEAD > .git/commit-id"
        def commit_id = readFile('.git/commit-id').trim()
        println commit_id
    
        stage "build"
        def app = docker.build "RepoTrain"
    
        stage "publish"
        app.push 'master'
        app.push "${commit_id}"
    }
}
